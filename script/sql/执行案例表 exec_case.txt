CREATE TABLE IF NOT EXISTS `exec_case` (
  `id` varchar(32) NOT NULL COMMENT '编号',
  `batch` int(4) NOT NULL COMMENT '执行批次',
  `case_desc` varchar(128) NOT NULL COMMENT '案例描述',
  `params` varchar(1024) NOT NULL COMMENT '变量数据',
  `is_used` varchar(1) NOT NULL COMMENT '是否弃用',
  `test_case_id` varchar(32) NOT NULL COMMENT '关联测试案例编号',
  `pre_exec_case_id` varchar(32) DEFAULT NULL COMMENT '前置执行案例编号',
  `is_archived` varchar(1) NOT NULL COMMENT '是否归档',
  `exec_times` int(4) NOT NULL COMMENT '执行次数',
  `last_time` datetime NOT NULL COMMENT '最后执行时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='执行案例表';
